27 void* storage_connect(const char *hostname, const int port)
28 {
29  // Create a socket.
30  int sock = socket(PF_INET, SOCK_STREAM, 0);
31  if (sock < 0)
32  return NULL;
33 
34  // Get info about the server.
35  struct addrinfo serveraddr, *res;
36  memset(&serveraddr, 0, sizeof serveraddr);
37  serveraddr.ai_family = AF_UNSPEC;
38  serveraddr.ai_socktype = SOCK_STREAM;
39  char portstr[MAX_PORT_LEN];
40 
41  //First place where I made logger call.
42  if(LOGGING == 1)
43  {
44  snprintf(portstr, sizeof portstr, "%d", port);
45  logger(stdout,portstr);
46  }
47  else if(LOGGING == 2)
48  {
49  snprintf(portstr, sizeof portstr, "%d", port);
50  logger(ClientFile,portstr);
51  }
52 
53  int status = getaddrinfo(hostname, portstr, &serveraddr, &res);
54  if (status != 0)
55  return NULL;
56 
57  // Connect to the server.
58  status = connect(sock, res->ai_addr, res->ai_addrlen);
59  if (status != 0)
60  return NULL;
61 
62  return (void*) sock;
63 }
